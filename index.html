create table if not exists kantoku_nippou(
  id uuid primary key default gen_random_uuid(),
  koji_no text, koji_name text, report_date date,
  data jsonb, created_at timestamptz default now());
alter table kantoku_nippou enable row level security;
create policy "all kn" on kantoku_nippou for all using(true) with check(true);

create table if not exists app_config(
  id text primary key, value jsonb);
alter table app_config enable row level security;
create policy "all cfg" on app_config for all using(true) with check(true);

create table if not exists koji_master(
  id uuid primary key default gen_random_uuid(),
  koji_no text, koji_name text,
  data jsonb, created_at timestamptz default now());
alter table koji_master enable row level security;
create policy "all km" on koji_master for all using(true) with check(true);
